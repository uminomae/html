21// header-field = field-name ":" OWS field-value OWS
22// ex. Host: 192.168.10.140
23// ex. Transfer-Encoding: chunked
24// ex. Connection: close
25void ParsingState::parseRequestHeader(HttpTransaction& httpTransaction) {
26 std::istringstream& reqIss = httpTransaction.getRequestStream();
27 std::string header;
28
29 while (std::getline(reqIss, header)) {
30 // 空行
31 if (header.empty() || header == "\r"){
32 break;
33 }
34
35 // ex. "Host: example.com "
36 size_t separatorPos = header.find(':');
37 // if ':'が見つかった場合
38 if (separatorPos != std::string::npos) {
39 // ':'の前後をsubstr()
40 // ex. "Host"
41 std::string key = header.substr(0, separatorPos);
42 // ex. " example.com "
43 std::string value = header.substr(separatorPos + 1);
44
45 // 値の前後の空白を削除 ex. " example.com "
46 // 前から空白以外の文字までを削除
47 size_t start = value.find_first_not_of(" \t\r\n");
48 if (start != std::string::npos) {
49 value = value.substr(start);
50 }
51 // 後ろから空白以外の文字までを削除
52 size_t end = value.find_last_not_of(" \t\r\n");
53 if (end != std::string::npos) {
54 value = value.substr(0, end + 1);
55 }
56
57 httpTransaction.getSocketState()._socketHeaders[key] = value;
58
59 if (key == "Transfer-Encoding" && value == "chunked"){
60 httpTransaction.getSocketState()._isChunk = true;
61
62#ifdef DEBUG_CHUNK
63putStrPair(__func__, "(): start", PINK190);
64putStrPair("chunk isChunk(1 is true): ", httpTransaction.getSocketState()._isChunk, PINK190);
65debugPutMap(httpTransaction.getSocketState()._socketHeaders);
66#endif
67
68 }
69 }
70 }
71}

89void ParsingState::paraseRequestChunk(
90 std::string& body,
91 SocketState& socketState)
92{
93#ifdef DEBUG_CHUNK
94putStrPair(__func__, "(): ▶️ start", PINK183);
95#endif
96
97 std::istringstream issBody(body);
98 std::string line;
99 int chunkSize = 0;
100
101 while (getline(issBody, line)){
102 std::istringstream iss(line);
103 // 16進数から10進数に変換
104 iss >> std::hex >> chunkSize;
105
106 if (chunkSize > 0){
107 // 指定サイズでcharの配列を確保
108 std::vector<char> buf(chunkSize);
109 // bufにサイズ分を読み込み
110 issBody.read(&buf[0], chunkSize);
111 // char[]からstring型に変換
112 std::string chunkData(buf.begin(), buf.end());
113 // "CRLF"2文字を無視
114 issBody.ignore(2);
115
116 // HttpTransactionクラスが毎回破棄されるため、状態をSocketStateが持つ
117 socketState._chunkSize += chunkSize;
118 socketState._chunkBuffer += chunkData;
119
120#ifdef DEBUG_CHUNK
121putStrPair(__func__, "(): 🔁 now", PINK183);
122putStrPair("chunkSize: \t\t", chunkSize, PINK183);
123putStrPair("chunkData: \t\t", chunkData, PINK183);
124putStrPair("socketState._chunkBuffer: ", socketState._chunkBuffer, PINK183);
125putStrPair("socketState._chunkSize: ", socketState._chunkSize, PINK183);
126#endif
127
128 } else if (chunkSize == 0) {
129 socketState._isChunk = false;
130 break;
131 } else {
132 throw std::runtime_error("Chunk size failed");
133 }
134
135 }
136 // 呼び出し元の変数にコピー
137 body = socketState._chunkBuffer;
138}

252// data in the path component (Section 3.3), serves to identify a
253// resource within the scope of the URI's scheme and naming authority
254// (if any). The query component is indicated by the first question
255// mark ("?") character and terminated by a number sign ("#") character
256// or by the end of the URI.
257
258// クエリコンポーネントは、階層的でないデータを含み、パスコンポーネント（セクション3.3）のデータとともに、URIのスキームや命名権限（もしあれば）の範囲内でリソースを識別するために使われます。クエリコンポーネントは、最初の疑問符（"?"）で示され、ナンバーサイン（"#") またはURIの終わりによって終了されます。

260// query = *( pchar / "/" / "?" )

262// The characters slash ("/") and question mark ("?") may represent data
263// within the query component. Beware that some older, erroneous
264// implementations may not handle such data correctly when it is used as
265// the base URI for relative references (Section 5.1), apparently
266// because they fail to distinguish query data from path data when
267// looking for hierarchical separators. However, as query components
268// are often used to carry identifying information in the form of
269// "key=value" pairs and one frequently used value is a reference to
270// another URI, it is sometimes better for usability to avoid percent-
271// encoding those characters.
272
273// スラッシュ（"/"）と疑問符（"?"）は、クエリコンポーネント内のデータを表すことができます。しかし、古い、誤った実装の一部では、これらのデータを相対参照の基本URIとして使用するときに、正しく処理されないことに注意してください（セクション5.1）。これは、階層的な区切りを探すときに、クエリデータとパスデータを区別できないためのようです。しかし、クエリコンポーネントはしばしば "key=value" の形式で識別情報を運ぶために使用され、よく使われる値の一つに他のURIへの参照があるため、これらの文字をパーセントエンコーディングで回避することが使いやすさのために時には良いでしょう。
