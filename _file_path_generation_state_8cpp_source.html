3// identifyConnectedServer(): クライアントのソケットが接続している ServerDirective の参照を返す
4// ※複雑になっている理由：returnは参照で取りたい。ポインタは避けたい。
5// ・ポインタ変数で扱うのが好ましいと考えるが、CとC++の理解の片方があれば読めるコードにしたい。
6// ・参照はnullが入れられないので宣言時に別にオブジェクトが必要。つまり、defaultServerを見つけるまでnullを入れておくわけにはいかない。
7// ・クラス型の配列のイテレーターが何を指しているのかわかりづらい。&(*it),*it,itの違い。
8// ・ポインタ変数をreturnに使用する際に参照型に戻す方法を冗長にした。
9ServerDirective& FilePathGenerationState::identifyConnectedServer(
10 const HttpTransaction& httpTransaction)
11{
12 // クライアントが接続しているサーバーのポートを取得
13 int clientSocket = httpTransaction.getClientSocket();
14 int connectedServPort = httpTransaction.getPortMapPtr().getConnectedServicePort(clientSocket);
15 if (connectedServPort == -1){
16 throw std::runtime_error("Connected server not found");
17 }
18
19 // デフォルトサーバーのアドレスを保持するためのポインタ変数
20 ServerDirective* defaultServerPtr = NULL;
21 // vector _serverDirectives(末尾にsついてる方)内の各 ServerDirective(sついてない方) を調べる（つまり.confの複数のServerディレクティブ設定を調べる）
22 std::vector<ServerDirective>::iterator it;
23 it = httpTransaction.getServerDirectives().begin();
24 for (/* void */; it != httpTransaction.getServerDirectives().end(); ++it){
25 // デフォルトサーバーを見つけた場合、そのアドレスを保存しておく。
26 if (it->getIsDefaultServer()){
27 defaultServerPtr = &(*it);}
28 // 接続ポートに一致するサーバー設定を見つけた場合、その参照を返す
29 if (connectedServPort == it->getPort()){
30 return *it;}
31 }
32
33 // 見つからない場合、デフォルトサーバーの参照を返す。
34 if (defaultServerPtr != NULL){
35 // ポインタと参照の関係がわかりづらいので冗長に記述
36 ServerDirective& defaultServerRef = *defaultServerPtr;
37 return defaultServerRef;
38 } else {
39 // デフォルトサーバーが設定されていない場合は例外を投げる
40 throw std::runtime_error("No suitable ServerDirective found");
41 }
42}

44bool FilePathGenerationState::isLocationUripattern(
45 HttpTransaction& httpTransaction,
46 std::string pattern,
47 bool extensionPattern) const
48{
49 size_t uriLen = httpTransaction.getSocketState()._socketUri.size();
50 size_t patternLen = pattern.size();
51
52 // URIの長さがパターンより短い場合は、一致しない
53 if (uriLen < patternLen){
54 return false;}
55
56 // 拡張子パターンの場合、URIの末尾を比較
57 if (extensionPattern){
58 // ex. "cgi"
59 return httpTransaction.getSocketState()._socketUri.compare(uriLen - patternLen, patternLen, pattern) == 0;
60 } else {
61 // ディレクトリパターンの場合、URIの先頭を比較
62 // パターンにスラッシュが続くか、URIがパターンに完全に一致する場合に限り真を返す
63 // ex. /html or /html/
64 if (httpTransaction.getSocketState()._socketUri == pattern || httpTransaction.getSocketState()._socketUri == pattern + "/"){
65 return true;
66 }
67 // ex. /upload/subdir
68 if (httpTransaction.getSocketState()._socketUri.compare(0, patternLen, pattern) == 0){
69 return pattern[patternLen] == '/';
70 }
71 }
72
73 return false;
74}

76bool FilePathGenerationState::findMatchingLocationDirective(
77 HttpTransaction& httpTransaction,
78 const std::map<std::string, LocationDirective>::const_iterator& it)
79{
80 std::string key = it->first;
81 size_t dotPos = key.find(".");
82
83 // locationディレクティブのuri patternが
84 // スラッシュで始まり、拡張子がない場合(ディレクトリパターン)
85 if (key[0] == '/' && key.size() > 1 && dotPos == std::string::npos){
86 if (isLocationUripattern(httpTransaction, key, false)){
87 return true;
88 }
89
90 // ’.'が存在する場合（拡張子パターン）
91 } else if (dotPos != std::string::npos){
92 std::string extensionStr = key.substr(dotPos);
93 if (isLocationUripattern(httpTransaction, extensionStr, true)){
94 return true;
95 }
96 }
97 return false;
98}
